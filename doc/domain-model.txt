[<Note>;<Viite>;<Kommentar>]<0..*-appearsIn 1[<Person>;<Henkilö>;<Person>]
[<Note>;<Viite>;<Kommentar>]<0..*-appearsIn 1[<Place>;<Plats>;<Paikkatieto>]
[<Note>;<Viite>;<Kommentar>]<0..*-appearsIn 1[<Term>;<Termi>;<Term>]

// Dokumentti-viite
class DocumentNote { 
  document: Document
  localId // viiteen id dokumentissä
  longText
  svnRevision // SVN revisio
  deleted: boolean
  createdOn: num
  note: Note
  publishable: boolean // julkaistavissa
  printPosition
}

// Viite
class Note { 
  allEditors : User*
  comments: Comment*
  description
  format: Format
  lastEditedBy: User
  lemma
  lemmaMeaning
  person: Person
  place: Place
  sources // lähteet
  status: Status
  subtextSources // Subtekstin lähteet
  term: Term
  types: Type*
  editedOn: number
}

// Kommentti
class Comment { 
  note: Note/Term
  message
  username
  createdAt: timestamp
}

// Henkilö
class Person { 
  normalizedForm: NameForm
  otherForms: NameForm*
  timeofBirth
  timeOfDeath
}

// Paikka
class Place { 
  normalizedForm: NameForm
  otherForm: NameForm*
}

// Termi, Käsite
class Term {
  basicForm // perusmouto
  meaning // merkitys
  language // kieli
  comments: Comment*
  description
  lastEditedBy: User
  sources // lähteet
  status: Status
  types: Type*  
}

// Nimimuoto
class NameForm {
  description
  first
  last
}

// Viitteen formaatti
enum Format {
  NOTE, // Viite
  PLACE, // Paikkaviite
  PERSON // Henkilöviite
}

// Viitteen tila
enum Status {
  INITIAL, // alustava
  DRAFT, // luonnos
  FINISHED // valmis
}

// Viitteen tyyppi
enum Type {
  WORD_EXPLANATION, // Sananselitys
  LITERARY, // Kirjallinen
  HISTORIAL, // Historiallinen
  DICTUM, // Sananparsi
  CRITIQUE, // Tekstikriittinen
  TITLE, 
  TRANSLATION,
  REFERENCE
}
