// Dokumentti-viite
class DocumentNote { 
  document: Document
  localId // viiteen id dokumentissä
  longText
  svnRevision // SVN revisio
  deleted: boolean
  createdOn: num
  note: Note
  publishable: boolean // julkaistavissa
}

// Viite
class Note { 
  allEditors : User*
  comments: NoteComment*
  description
  format: NoteFormat
  lastEditedBy: User
  lemma
  lemmaMeaning
  person: Person
  place: Place
  sources // lähteet
  status: NoteStatus
  subtextSources // Subtekstin lähteet
  term: Term
  types: NoteType*
  editedOn: number
}

// Kommentti
class NoteComment { 
  note: Note
  message
  username
  createdAt: timestamp
}

// Henkilö
class Person { 
  normalizedForm: NameForm
  otherForms: NameForm*
  timeofBirth
  timeOfDeath
}

// Paikka
class Place { 
  normalizedForm: NameForm
  otherForm: NameForm*
}

// Termi, Käsite
class Term {
  basicForm // perusmouto
  meaning // merkitys
  language // kieli
}

// Nimimuoto
class NameForm {
  description
  first
  last
}

// Viitteen formaatti
enum NoteFormat {
  NOTE, // Viite
  PLACE, // Paikkaviite
  PERSON // Henkilöviite
}

// Viitteen tila
enum NoteStatus {
  INITIAL, // alustava
  DRAFT, // luonnos
  FINISHED // valmis
}

// Viitteen tyyppi
enum NoteType {
  WORD_EXPLANATION, // Sananselitys
  LITERARY, // Kirjallinen
  HISTORIAL, // Historiallinen
  DICTUM, // Sananparsi
  CRITIQUE // Tekstikriittinen
}
